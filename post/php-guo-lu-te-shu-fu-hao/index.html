PHP 过滤特殊符号

· 2022-03-02 Wednesday · # php

过滤特殊字符，仅保留中文、字母、数字、下划线

<?php

// 过滤特殊字符，仅保留中文、字母、数字、下划线
function match_chinese($chars, $encoding = 'utf8')
{
    $pattern = ($encoding == 'utf8') ? '/[\x{4e00}-\x{9fa5}a-zA-Z0-9_]/u' : '/[\x80-\xFF]/';
    preg_match_all($pattern, $chars, $result);
    $temp = join('', $result[0]);
    return $temp;
}
